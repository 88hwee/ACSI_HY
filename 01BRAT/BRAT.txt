<script>
/* ----- Tab switching (no arrow functions) ----- */
var tabButtons = document.getElementsByClassName("tabBtn");
var sections   = document.getElementsByClassName("gameSection");

function showSection(id){
  // activate correct tab
  for(var i=0;i<tabButtons.length;i++){
    var btn = tabButtons[i];
    if(btn.getAttribute("data-target") === id){
      btn.classList.add("active");
    }else{
      btn.classList.remove("active");
    }
  }
  // show correct section
  for(var j=0;j<sections.length;j++){
    var sec = sections[j];
    if(sec.id === id){
      sec.classList.add("active");
    }else{
      sec.classList.remove("active");
    }
  }
}

for(var i=0;i<tabButtons.length;i++){
  tabButtons[i].addEventListener("click", function(){
    var targetId = this.getAttribute("data-target");
    showSection(targetId);
  });
}

/* ----- Generic checker for each BRAT block ----- */
function attachChecker(buttonId, feedbackId, sectionId){
  var btn = document.getElementById(buttonId);
  var fb  = document.getElementById(feedbackId);
  var sec = document.getElementById(sectionId);

  btn.addEventListener("click", function(){
    var selects = sec.querySelectorAll("select[data-answer]");
    var total = selects.length;
    var correct = 0;
    var blank = 0;

    selects.forEach(function(sel){
      sel.classList.remove("correct","incorrect");
      var ans = sel.getAttribute("data-answer");
      var val = sel.value;
      if(!val){
        blank++;
        return;
      }
      if(val === ans){
        correct++;
        sel.classList.add("correct");
      }else{
        sel.classList.add("incorrect");
      }
    });

    if(blank > 0){
      fb.textContent = "Fill in all blanks first, then check again üôÇ";
    }else{
      fb.textContent = "You got " + correct + " out of " + total + " blanks correct.";
    }
  });
}

attachChecker("checkB","feedbackB","gameB");
attachChecker("checkR","feedbackR","gameR");
attachChecker("checkA","feedbackA","gameA");
attachChecker("checkT","feedbackT","gameT");
</script>
