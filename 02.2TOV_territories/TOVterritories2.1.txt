<script>
const territoryData = [
    {name: "Alsace-Lorraine", lost: true, info: "Returned to France. Germany had seized this region in 1871, and France made regaining it a primary war goal."},
    {name: "Polish Corridor", lost: true, info: "Given to the new state of Poland to provide them access to the Baltic Sea. This separated East Prussia from the rest of Germany."},
    {name: "Saar Basin", lost: true, info: "Rich in coal, this area was placed under League of Nations control for 15 years, with its coal output going to France."},
    {name: "Rhineland", lost: false, info: "Germany KEPT this land, but it was 'demilitarized'. No German troops were allowed near the French border."},
    {name: "Danzig (Gda≈Ñsk)", lost: true, info: "Became a 'Free City' under League of Nations protection to give Poland port access, though it was historically German."},
    {name: "Sudetenland", lost: false, info: "Germany did NOT lose this in 1919. It was part of the new Czechoslovakia. Hitler later annexed it in 1938."},
    {name: "German Colonies", lost: true, info: "Germany lost ALL overseas colonies in Africa and the Pacific (e.g., Togo, Cameroon, Namibia). They became 'mandates'."},
    {name: "Bavaria", lost: false, info: "Bavaria remained an integral state within the German Weimar Republic."},
    {name: "Northern Schleswig", lost: true, info: "Returned to Denmark following a public vote (plebiscite) by the people living there."},
    {name: "Belgium", lost: false, info: "Belgium was never German territory, though Germany occupied it during the war. Germany withdrew its troops."},
    {name: "Eupen-Malm√©dy", lost: true, info: "A small border region given to Belgium as reparations for war damage."},
    {name: "Memel Territory", lost: true, info: "A strip of land on the East Prussian border, eventually given to Lithuania."}
];

class TOVGame {
    constructor() {
        this.selectedTiles = new Set();
        this.gameChecked = false;
        this.container = document.getElementById("gameContainer");
        this.modal = document.getElementById("modalOverlay");
        this.init();
    }
    
    init() {
        this.createTiles();
        this.bindEvents();
        this.updateStats();
        // Initialize total lost count dynamically
        document.getElementById("totalLost").textContent = territoryData.filter(t => t.lost).length;
    }
    
    createTiles() {
        const tilesDiv = document.getElementById("tiles");
        tilesDiv.innerHTML = '';
        
        territoryData.forEach((territory, index) => {
            const tile = document.createElement("div");
            tile.className = "tile";
            
            // Inner content with badge
            tile.innerHTML = `
                ${territory.name}
                <span class="info-badge">?</span>
            `;
            
            tile.addEventListener("click", () => this.handleTileClick(tile, index));
            tilesDiv.appendChild(tile);
        });
    }
    
    handleTileClick(tile, index) {
        if (this.gameChecked) {
            // GAME OVER: Show Info Modal
            this.showInfo(index);
        } else {
            // GAME ACTIVE: Toggle Selection
            this.toggleSelection(tile, index);
        }
    }
    
    toggleSelection(tile, index) {
        if (this.selectedTiles.has(index)) {
            this.selectedTiles.delete(index);
            tile.classList.remove("selected");
        } else {
            this.selectedTiles.add(index);
            tile.classList.add("selected");
        }
        this.updateStats();
    }
    
    showInfo(index) {
        const data = territoryData[index];
        const mTitle = document.getElementById("mTitle");
        const mStatus = document.getElementById("mStatus");
        const mText = document.getElementById("mText");
        
        mTitle.textContent = data.name;
        mText.textContent = data.info;
        
        if (data.lost) {
            mStatus.textContent = "LOST by Germany";
            mStatus.className = "modal-status status-lost";
        } else {
            mStatus.textContent = "KEPT by Germany";
            mStatus.className = "modal-status status-kept";
        }
        
        this.modal.classList.add("active");
    }
    
    updateStats() {
        document.getElementById("selectedCount").textContent = this.selectedTiles.size;
    }
    
    checkAnswer() {
        if (this.gameChecked) return;
        
        this.gameChecked = true;
        this.container.classList.add("game-over");
        
        const tiles = document.querySelectorAll(".tile");
        let correct = 0;
        let incorrect = 0;
        let missed = 0;
        
        tiles.forEach((tile, index) => {
            const isSelected = this.selectedTiles.has(index);
            const shouldBeSelected = territoryData[index].lost;
            
            // Remove selection look for visual clarity in results
            tile.classList.remove("selected");
            
            if (isSelected && shouldBeSelected) {
                tile.classList.add("correct");
                correct++;
            } else if (isSelected && !shouldBeSelected) {
                tile.classList.add("incorrect");
                incorrect++;
            } else if (!isSelected && shouldBeSelected) {
                tile.classList.add("missed");
                missed++;
            }
        });
        
        const totalLost = territoryData.filter(t => t.lost).length;
        const accuracy = Math.round((correct / totalLost) * 100);
        
        document.getElementById("accuracy").textContent = `${accuracy}%`;
        
        this.showResult(correct, totalLost, accuracy);
    }
    
    showResult(correct, totalLost, accuracy) {
        const resultDiv = document.getElementById("result");
        resultDiv.classList.add("show");
        
        let message = "Keep studying! üìö";
        if (accuracy === 100) message = "Perfect! Historical Expert! üèÜ";
        else if (accuracy >= 70) message = "Great job! Almost there! üëç";
        
        resultDiv.innerHTML = `
            <div><strong>${message}</strong> (Score: ${accuracy}%)</div>
            <span class="click-hint">üëâ Click any box to learn why it was Lost or Kept</span>
        `;
    }
    
    resetGame() {
        this.selectedTiles.clear();
        this.gameChecked = false;
        this.container.classList.remove("game-over");
        
        // Reset Visuals
        const tiles = document.querySelectorAll(".tile");
        tiles.forEach(tile => {
            tile.className = "tile";
        });
        
        document.getElementById("result").classList.remove("show");
        document.getElementById("accuracy").textContent = "-";
        this.updateStats();
    }
    
    bindEvents() {
        document.getElementById("checkBtn").addEventListener("click", () => this.checkAnswer());
        document.getElementById("resetBtn").addEventListener("click", () => this.resetGame());
        
        // Modal Closing Logic
        document.getElementById("closeModal").addEventListener("click", () => {
            this.modal.classList.remove("active");
        });
        
        // Close modal on outside click
        this.modal.addEventListener("click", (e) => {
            if (e.target === this.modal) {
                this.modal.classList.remove("active");
            }
        });
        
        // Close on Escape key
        document.addEventListener("keydown", (e) => {
            if (e.key === "Escape" && this.modal.classList.contains("active")) {
                this.modal.classList.remove("active");
            }
        });
    }
}

// Initialize the game
new TOVGame();
</script>
