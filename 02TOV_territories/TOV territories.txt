<script>
const data = [
  {name:"Alsace-Lorraine",lost:true},
  {name:"Polish Corridor",lost:true},
  {name:"Saar (coalfields under League for 15 years)",lost:true},
  {name:"Rhineland (completely given away)",lost:false}, // only demilitarised
  {name:"Danzig (Free City)",lost:true},
  {name:"Sudetenland",lost:false},
  {name:"German colonies in Africa",lost:true},
  {name:"All of Bavaria",lost:false},
  {name:"Northern Schleswig to Denmark",lost:true},
  {name:"Belgium",lost:false}
];

const tilesDiv=document.getElementById("tiles");
data.forEach((item,idx)=>{
  const d=document.createElement("div");
  d.className="tile";
  d.textContent=item.name;
  d.dataset.index=idx;
  d.addEventListener("click",()=>d.classList.toggle("selected"));
  tilesDiv.appendChild(d);
});

const res=document.getElementById("result");
document.getElementById("checkBtn").addEventListener("click",()=>{
  const tiles=document.querySelectorAll(".tile");
  let correct=0,wrong=0;
  tiles.forEach(t=>{
    const idx=+t.dataset.index;
    const selected=t.classList.contains("selected");
    const should=data[idx].lost;
    if(selected && should) correct++;
    if(selected && !should) wrong++;
  });
  const totalLost=data.filter(x=>x.lost).length;
  res.innerHTML=`You correctly selected <b>${correct}</b> out of <b>${totalLost}</b> lost territories.<br>
  Wrong selections: <b>${wrong}</b>.`;
});
</script>
